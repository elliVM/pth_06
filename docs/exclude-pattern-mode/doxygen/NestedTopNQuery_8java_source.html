124 public Table<Record> getTableStatement(Condition journaldbConditionArg, Date day) {
125 logger.debug("NestedTopNQuery.getTableStatement called condition <{}> date <{}>", journaldbConditionArg, day);
126 SelectOnConditionStep<Record> selectOnConditionStep = DSL
127 .select(resultFields)
128 .from(GetArchivedObjectsFilterTable.FILTER_TABLE)
129 .innerJoin(JOURNALDB.LOGFILE.forceIndex(Indexes.LOGFILE_CIX_LOGFILE_LOGDATE_HOST_ID_LOGTAG.getName()))
130 .on(JOURNALDB.LOGFILE.HOST_ID.eq(GetArchivedObjectsFilterTable.host_id).and(JOURNALDB.LOGFILE.LOGTAG.eq(GetArchivedObjectsFilterTable.tag)));
131
132 if (excludePatternEnabled && streamDBClient.bloomEnabled()) {
133 logger.info("Exclude pattern mode enabled using pattern <[{}]>", excludePattern);
134 final QueryCondition stringEqualsCondition = new StringEqualsCondition(
135 excludePattern,
136 BLOOMDB.FILTERTYPE.PATTERN
137 );
138 final ConditionMatchBloomDBTables matchBloomDBTables = new ConditionMatchBloomDBTables(
139 streamDBClient.ctx(),
140 stringEqualsCondition
141 );
142 // join all tables that match the pattern to be excluded, so they can be filtered
143 for (final Table<?> table : matchBloomDBTables.tables()) {
144 logger.debug("filtering table <{}>", table);
145 selectOnConditionStep = selectOnConditionStep
146 .leftJoin(table)
147 .on(JOURNALDB.LOGFILE.ID.eq(table.field("partition_id", ULong.class)))
148 .leftJoin(BLOOMDB.FILTERTYPE)
149 .on(BLOOMDB.FILTERTYPE.ID.eq(table.field("filter_type_id", ULong.class)).and(BLOOMDB.FILTERTYPE.PATTERN.eq(excludePattern)));
150 journaldbConditionArg = journaldbConditionArg.and(BLOOMDB.FILTERTYPE.ID.isNull());
151 }
152 logger.debug("updated journaldb condition <{}>", journaldbConditionArg);
153 }
154 else if (streamDBClient.bloomEnabled()) {
155 // join all tables needed for the condition generated by walker
156 final Set<Table<?>> tables = streamDBClient.walker().conditionRequiredTables();
157 if (!tables.isEmpty()) {
158 for (final Table<?> table : tables) {
159 if (logger.isInfoEnabled()) {
160 logger.info("Left join pattern match table: <{}>", table.getName());
161 }
162 selectOnConditionStep = selectOnConditionStep
163 .leftJoin(table)
164 .on(JOURNALDB.LOGFILE.ID.eq((Field<ULong>) table.field("partition_id")));
165 }
166 }
167 }
168
169 logger.debug("NestedTopNQuery.getTableStatement exit");
170 return selectOnConditionStep
171 .where(JOURNALDB.LOGFILE.LOGDATE.eq(day).and(journaldbConditionArg))
172 .orderBy(logtimeForOrderBy, JOURNALDB.LOGFILE.ID.asc())
173 .asTable(innerTable);
174 }
