103 public ArchiveConfig(Map<String, String> opts) {
104 s3EndPoint = getOrThrow(opts, "S3endPoint");
105 s3Identity = getOrThrow(opts, "S3identity");
106 s3Credential = getOrThrow(opts, "S3credential");
107 dbUsername = getOrThrow(opts, "DBusername");
108 dbPassword = getOrThrow(opts, "DBpassword");
109 dbUrl = getOrThrow(opts, "DBurl");
110 bloomEnabled = opts.getOrDefault("bloom.enabled", "false").equalsIgnoreCase("true");
111 withoutFilters = opts.getOrDefault("bloom.withoutFilters", "false").equalsIgnoreCase("true");
112 withoutFiltersPattern = opts.getOrDefault("bloom.withoutFiltersPattern", "");
113 bloomDbName = opts.getOrDefault("DBbloomdbname", "bloomdb");
114
115 dbJournalDbName = opts.getOrDefault("DBjournaldbname", "journaldb");
116 dbStreamDbName = opts.getOrDefault("DBstreamdbname", "streamdb");
117
118 // hide exceptions
119 hideDatabaseExceptions = opts.getOrDefault("hideDatabaseExceptions", "false").equalsIgnoreCase("true");
120
121 // skip not rfc5424 parseable files
122 skipNonRFC5424Files = opts.getOrDefault("skipNonRFC5424Files", "false").equalsIgnoreCase("true");
123
124 archiveIncludeBeforeEpoch = Long
