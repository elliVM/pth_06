135 public Config(Map<String, String> opts) {
136 this.query = opts.get("queryXML");
137 if (this.query == null) {
138 throw new IllegalArgumentException("no queryXML provided");
139 }
140
141 batchConfig = new BatchConfig(opts);
142 auditConfig = new AuditConfig(opts);
143
144 isArchiveEnabled = opts.getOrDefault("archive.enabled", "false").equalsIgnoreCase("true");
145 if (isArchiveEnabled) {
146 archiveConfig = new ArchiveConfig(opts);
147 }
148 else {
149 archiveConfig = new ArchiveConfig();
150 }
151
152 isKafkaEnabled = opts.getOrDefault("kafka.enabled", "false").equalsIgnoreCase("true");
153 if (isKafkaEnabled) {
154 kafkaConfig = new KafkaConfig(opts);
155 }
156 else {
157 kafkaConfig = new KafkaConfig();
158 }
159
160 // check that at least one datasource is enabled
161 if (!isArchiveEnabled && !isKafkaEnabled) {
162 throw new IllegalStateException("No datasources enabled");
163 }
164
165 // fetch metadata (defaults to false)
166 isMetadataQuery = opts.getOrDefault("metadataQuery.enabled", "false").equalsIgnoreCase("true");
167 }
