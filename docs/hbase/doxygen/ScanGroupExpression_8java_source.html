96 public List<ScanRange> value() {
97 final List<XMLValueExpression> indexList = new ArrayList<>();
98 final List<XMLValueExpression> hostList = new ArrayList<>();
99 final List<XMLValueExpression> sourceTypeList = new ArrayList<>();
100 final List<Long> earliestList = new ArrayList<>();
101 final List<Long> latestList = new ArrayList<>();
102 final FilterList filterList = new FilterList();
103
104 for (final Expression child : expressions) {
105 if (child.isLeaf()) {
106 final Expression.Tag tag = child.tag();
107 final XMLValueExpression xmlValueExpression = (XMLValueExpression) child.asLeaf();
108 final String value = xmlValueExpression.value().replace("*", ".*"); // regex wildcard
109 final CompareOperator operator;
110 final String operation = xmlValueExpression.operation();
111 if ("EQUALS".equalsIgnoreCase(operation)) {
112 operator = CompareOperator.EQUAL;
113 }
114 else if ("NOT_EQUALS".equalsIgnoreCase(operation)) {
115 operator = CompareOperator.NOT_EQUAL;
116 }
117 else {
118 throw new IllegalArgumentException("Unsupported operation <" + operation + ">");
119 }
120 switch (tag) {
121 case INDEX:
122 indexList.add(xmlValueExpression);
123 break;
124 case SOURCETYPE:
125 // for SQL condition
126 sourceTypeList.add(xmlValueExpression);
127 // HBase filter
128 final Filter sourceTypeFilter = new SingleColumnValueFilter(
129 Bytes.toBytes("meta"), // column family
130 Bytes.toBytes("s"), // stream
131 operator,
132 new RegexStringComparator(value)
133 );
134 filterList.addFilter(sourceTypeFilter);
135 break;
136 case HOST:
137 // for SQL condition
138 hostList.add(xmlValueExpression);
139 // HBase filter
140 final Filter hostFilter = new SingleColumnValueFilter(
141 Bytes.toBytes("meta"), // column family
142 Bytes.toBytes("h"), // host
143 operator,
144 new RegexStringComparator(value)
145 );
146 filterList.addFilter(hostFilter);
147 break;
148 case EARLIEST:
149 earliestList.add(Long.valueOf(xmlValueExpression.value()));
150 break;
151 case LATEST:
152 latestList.add(Long.valueOf(xmlValueExpression.value()));
153 break;
154 default:
155 throw new IllegalArgumentException("Unsupported leaf tag <" + tag + ">");
156 }
157 }
158 }
159 if (earliestList.isEmpty() || latestList.isEmpty()) {
160 throw new IllegalStateException("Scan group did not have required time qualifiers");
161 }
162 else if ((earliestList.size() > 1 || latestList.size() > 1) && LOGGER.isWarnEnabled()) {
163 LOGGER
164 .warn(
165 "Multiple time qualifiers found. earliest size: <{}> latest size: <{}>",
166 earliestList.size(), latestList.size()
167 );
168 }
169 final Long earliest = Collections.min(earliestList);
170 final Long latest = Collections.max(latestList);
171
172 final List<List<Long>> streamIDs = indexList
173 .stream()
174 // maps index expressions to StreamIDs objects
175 .map(xmlExpression -> new StreamIDs(ctx, new StreamDBCondition(xmlExpression, hostList, sourceTypeList).condition())).map(StreamIDs::streamIdList).collect(Collectors.toList());
176 LOGGER.info("Stream ID list <{}>", streamIDs);
177
178 final List<ScanRange> scanRangeImpls = streamIDs
179 .stream()
180 // map stream ids to a list of ScanRanges
181 .map(
182 streamIDList -> streamIDList
183 .stream()
184 .map(streamId -> new ScanRangeImpl(streamId, earliest, latest, filterList))
185 .collect(Collectors.toList())
186 )
187 // flatten resulting scan range lists into a single list
188 .flatMap(List::stream)
189 .collect(Collectors.toList());
190 if (LOGGER.isInfoEnabled()) {
191 LOGGER
192 .info(
193 "Created <{}> scan ranges with earliest <{}> to latest <{}>", scanRangeImpls.size(),
194 earliest, latest
195 );
196 }
197 final MergeIntersectingRanges mergeIntersectingRanges = new MergeIntersectingRanges(scanRangeImpls);
198 return mergeIntersectingRanges.mergedRanges();
199 }
