153 @Override
154 public List<Result> nextWindow(final long duration) throws IOException {
155 if (scanRange.isStub()) {
156 throw new IllegalStateException("ScanRange was stub");
157 }
158 if (!isOpen) {
159 throw new IllegalStateException("RangeView was closed");
160 }
161 if (duration < 1) {
162 throw new IllegalArgumentException("given duration <" + duration + "> was not a positive");
163 }
164
165 final long windowEnd = currentEpoch + duration;
166 final List<Result> results = new ArrayList<>();
167 while (true) {
168 final Result next;
169 // check if buffer has a stored result
170 if (bufferedResult != null) {
171 final long bufferedResultRowKeyEpoch = new EpochFromRowKey(bufferedResult.getRow()).epoch();
172 boolean bufferedIsWithinWindow = bufferedResultRowKeyEpoch >= currentEpoch
173 && bufferedResultRowKeyEpoch < currentEpoch + duration;
174 if (bufferedIsWithinWindow) {
175 next = bufferedResult;
176 bufferedResult = null;
177 }
178 else {
179 break;
180 }
181 }
182 else {
183 next = resultScanner.next();
184 }
185 if (next == null) {
186 isFinished = true;
187 break;
188 }
189 final long rowEpoch = new EpochFromRowKey(next.getRow()).epoch();
190
191 if (rowEpoch >= windowEnd) {
192 // store result that passed window end to buffer
193 bufferedResult = next;
194 break;
195 }
196
197 results.add(next);
198 }
199 currentEpoch += duration;
200 return results;
201 }
