61 public ScanRangeImpl(final long streamId, final long earliest, final long latest, final FilterList filterList) {
62 this.streamId = streamId;

64 this.latest = latest;
65 this.filterList = filterList;

68 @Override
69 public Scan toScan() {
70 final ByteBuffer startBuffer = ByteBuffer.allocate(16); // first 2 long values
71 startBuffer.putLong(streamId);
72 startBuffer.putLong(earliest);
73 final ByteBuffer stopBuffer = ByteBuffer.allocate(16); // first 2 long values
74 stopBuffer.putLong(streamId);
75 stopBuffer.putLong(latest);
76 final Scan scan = new Scan().withStartRow(startBuffer.array()).withStopRow(stopBuffer.array());
77 scan.setFilter(filterList);
78 return scan;
79 }

101 @Override
102 public ScanRange toRangeBetween(long earliestLimit, long latestLimit) {
103 final boolean rangeIntersects = new ScanRangeImpl(streamId, earliestLimit - 1, latestLimit + 1, filterList)
104 .intersects(this);
105 final ScanRange result;
106 if (rangeIntersects) {
107 long updatedEarliest = earliest;
108 long updatedLatest = latest;
109 if (earliestLimit > earliest) {
110 updatedEarliest = earliestLimit;
111 }
112 if (latestLimit < latest) {
113 updatedLatest = latestLimit;
114 }
115 if (updatedEarliest == updatedLatest) {
116 result = new StubScanRange();
117 }
118 else if (updatedEarliest > updatedLatest) {
119 result = new StubScanRange();
120 }
121 else {
122 result = new ScanRangeImpl(streamId, updatedEarliest, updatedLatest, filterList);
123 }
124 }
125 else {
126 result = new StubScanRange();
127 }
128 return result;
129 }

131 public boolean intersects(final ScanRange other) {
132 if (!Objects.equals(this.streamId, other.streamId()) || !filterList.equals(other.filterList())) {
133 return false;
134 }
135 return this.earliest <= other.latest() && other.earliest() <= this.latest;
136 }
137
138 public ScanRangeImpl merge(final ScanRange other) {
139 if (intersects(other)) {
140 final long minEarliest = Math.min(earliest, other.earliest());
141 final long maxLatest = Math.max(latest, other.latest());
142 return new ScanRangeImpl(streamId, minEarliest, maxLatest, filterList);
143 }
144 else {
145 throw new IllegalArgumentException("Unable to merge ranges did not intersect");
146 }
147 }
