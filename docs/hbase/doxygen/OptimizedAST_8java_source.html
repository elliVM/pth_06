79 public Expression transformed() {
80 Expression current = root;
81 Expression last;
82 LOGGER.info("Start AST:\n {}", new PrintAST(root).format());
83 int i = 0;
84 do { // apply until no optimization changes occur
85 last = current;
86 current = walkAndApply(current);
87 i++;
88 LOGGER.info("Optimize run <{}> AST:\n {}", i, new PrintAST(current).format());
89 }
90 while (!current.equals(last));
91 LOGGER.info("Final AST:\n {}", new PrintAST(current).format());
92 return current;
93 }
94
95 // recursively apply to all expressions in AST
96 private Expression walkAndApply(final Expression expression) {
97 final Expression.Tag tag = expression.tag();
98 final Expression result;
99 final List<Expression> children;
100 if (expression.isLogical()) {
101 children = expression.asLogical().children();
102 }
103 else {
104 children = Collections.emptyList();
105 }
106 switch (tag) {
107 case OR:
108 final List<Expression> optimizedOrChildren = new ArrayList<>();
109 for (final Expression child : children) {
110 final Expression optimized = walkAndApply(child);
111 optimizedOrChildren.add(optimized);
112 }
113 result = new OrExpression(optimizedOrChildren);
114 break;
115 case AND:
116 final List<Expression> optimizedAndChildren = new ArrayList<>();
117 for (final Expression child : children) {
118 final Expression optimized = walkAndApply(child);
119 optimizedAndChildren.add(optimized);
120 }
121 result = new AndExpression(optimizedAndChildren);
122 break;
123 case EARLIEST:
124 case LATEST:
125 case INDEX:
126 case INDEXSTATEMENT:
127 case HOST:
128 case SOURCETYPE:
129 case EMPTY:
130 result = expression;
131 break;
132 default:
133 throw new IllegalArgumentException("Unsupported tag <" + tag + ">");
134 }
135 return transformedSingleExpression(result);
136 }
137
138 // apply optimizations once for a single expression
139 private Expression transformedSingleExpression(final Expression expression) {
140 LOGGER.info("incoming Expression:\n {}", new PrintAST(expression).format());
141 Expression result = new UniqueChildren(expression).transformed();
142 LOGGER.info("Unique children:\n {}", new PrintAST(result).format());
143 result = new IdentitySimplification(result).transformed();
144 LOGGER.info("Identity simplification:\n {}", new PrintAST(result).format());
145 result = new PrunedInvalidTimeQualifier(result).transformed();
146 LOGGER.info("Pruned invalid time qualifier:\n {}", new PrintAST(result).format());
147 result = new EmptyPruned(result).transformed();
148 LOGGER.info("Empty pruned:\n {}", new PrintAST(result).format());
149 result = new FlattenLogical(result).transformed();
150 LOGGER.info("Logical flattened:\n {}", new PrintAST(result).format());
151 return result;
152 }
