75 public Expression asAST() {
76 final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
77 final DocumentBuilder builder;
78 try {
79 builder = factory.newDocumentBuilder();
80 }

83 }

87 }
88 catch (final IOException | SAXException e) {
89 throw new RuntimeException("Error parsing XML to elements: " + e.getMessage());
90 }
91 final Element root = document.getDocumentElement();
92 return elementToNode(root);
93 }

95 private Expression elementToNode(final Element element) {
96 final String tagName = element.getTagName();
97 LOGGER.info("Incoming element <{}>", tagName);
98 final Expression result;
99 switch (tagName.toLowerCase()) {
100 case "and":
101 final List<Expression> andExpMembers = visitLogical(element);
102 final Expression andLeft = andExpMembers.get(0);
103 final Expression andRight = andExpMembers.get(1);
104 result = new AndExpression(andLeft, andRight);
105 break;
106 case "or":
107 final List<Expression> expressions = visitLogical(element);
108 final Expression left = expressions.get(0);
109 final Expression right = expressions.get(1);
110 result = new OrExpression(left, right);
111 break;
112 case "index":
113 result = visitLeaf(element, Expression.Tag.INDEX);
114 break;
115 case "host":
116 result = visitLeaf(element, Expression.Tag.HOST);
117 break;
118 case "sourcetype":
119 result = visitLeaf(element, Expression.Tag.SOURCETYPE);
120 break;
121 case "earliest":
122 case "index_earliest":
123 result = visitLeaf(element, Expression.Tag.EARLIEST);
124 break;
125 case "latest":
126 case "index_latest":
127 result = visitLeaf(element, Expression.Tag.LATEST);
128 break;
129 case "indexstatement":
130 result = visitLeaf(element, Expression.Tag.INDEXSTATEMENT);
131 break;
132 default:
133 throw new IllegalArgumentException("Unsupported element <" + tagName + ">");
134 }
135 return result;
136 }
137
138 private Expression visitLeaf(final Element element, final Expression.Tag tag) {
139 final String value = element.getAttribute("value");
140 final String operation = element.getAttribute("operation");
141 return new XMLValueExpressionImpl(value, operation, tag);
142 }
143
144 private List<Expression> visitLogical(final Element element) {
145 final List<Expression> expressions = new ArrayList<>();
146 final NodeList childNodes = element.getChildNodes();
147 for (int i = 0; i < childNodes.getLength(); i++) {
148 final Node node = childNodes.item(i);
149 if (node instanceof Element) {
150 final Element nodeElement = (Element) node;
151 // recursive call
152 final Expression exp = elementToNode(nodeElement);
153 expressions.add(exp);
154 }
155 else {
156 throw new IllegalArgumentException("Element children contained a non Element node");
157 }
158 }
159 return expressions;
160 }
