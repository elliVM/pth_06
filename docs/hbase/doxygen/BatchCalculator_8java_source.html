82 public List<List<BatchUnit>> processRange(Offset start, Offset end) {
83 LOGGER.debug("processRange");
84
85 List<BatchUnit> slice = new ArrayList<>();
86
87 if (useHBase()) {
88 slice.addAll(new HBaseBatchSliceCollection(hbaseQuery).processRange(start, end));
89 }
90 else if (useArchive()) {
91 slice.addAll(new ArchiveRangeProcessor(archiveQuery).processRange(start, end));
92 }
93 if (useKafka()) {
94 slice.addAll(new KafkaRangeProcessor(kafkaQuery).processRange(start, end));
95 }
96
97 return buildBatch(slice);
98
99 }

101 private List<List<BatchUnit>> buildBatch(List<BatchUnit> sliceCollection) {
102
103 final PriorityQueue<BatchTaskQueue> runQueues = new PriorityQueue<>();
104
105 for (int i = 0; i < config.batchConfig.numPartitions; i++) {
106 runQueues.add(new BatchTaskQueue());
107 }
108
109 PriorityQueue<BatchUnit> batchUnitQueue = new PriorityQueue<>(sliceCollection);
110
111 while (!batchUnitQueue.isEmpty()) {
112 BatchUnit longestObject = batchUnitQueue.poll();
113
114 if (!runQueues.isEmpty()) {
115 BatchTaskQueue shortestQueue = runQueues.poll();
116 shortestQueue.add(longestObject);
117 runQueues.add(shortestQueue);
118 }
119 else {
120 throw new IllegalStateException("number of partitions is less than 1");
121 }
122 }
123
124 final List<List<BatchUnit>> taskSliceQueues = new ArrayList<>(runQueues.size());
125
126 for (BatchTaskQueue btq : runQueues) {
127 taskSliceQueues.add(btq.getQueue());
128 }
129
130 LOGGER.debug("getBatch <{}> ", taskSliceQueues);
131 return taskSliceQueues;
132 }
133
