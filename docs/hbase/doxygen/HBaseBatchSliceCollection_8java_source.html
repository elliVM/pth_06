125 public HBaseBatchSliceCollection processRange(final Offset start, final Offset end) {
126 this.clear(); // clear internal list
127 final long startOffsetLong = ((DatasourceOffset) start).getArchiveOffset().offset();
128 final long endOffsetLong = ((DatasourceOffset) end).getArchiveOffset().offset();
129 final long lastCommited = hBaseQuery.latest();
130 final long selectedStartOffset = Math.max(startOffsetLong, lastCommited);
131 LOGGER.debug("processRange() start <{}> end <{}>", selectedStartOffset, endOffsetLong);
132 final SynchronizedHourlyResults synchronizedHourlyResults = new SynchronizedHourlyResults(
133 hBaseQuery,
134 selectedStartOffset
135 );
136 final List<Result> results = new ArrayList<>();
137 final long maxWeight = (long) quantumLength * numPartitions;
138
139 while (synchronizedHourlyResults.hasNext()) {
140
141 final List<Result> hourlyResults = synchronizedHourlyResults.nextHour();
142 final long currentEpoch = synchronizedHourlyResults.currentEpoch();
143
144 if (!hourlyResults.isEmpty()) {
145 final BatchSizeLimit batchSizeLimit = new BatchSizeLimit(maxWeight, totalObjectCountLimit);
146 for (final Result hourlyResult : hourlyResults) {
147 final long fileSize = Bytes.toLong(hourlyResult.getValue(meta, Bytes.toBytes("fs")));
148 final float hourlyResultEstimatedFileSize = fileSize * fileCompressionRatio / 1024 / 1024
149 / processingSpeed;
150 batchSizeLimit.add(hourlyResultEstimatedFileSize);
151 if (!batchSizeLimit.isOverLimit()) {
152 results.add(hourlyResult);
153 }
154 else {
155 LOGGER
156 .info(
157 "Hourly results were over batch size limit with <{}> logfiles, ignoring rest of the results",
158 hourlyResult.size()
159 );
160 break;
161 }
162 }
163 }
164 hBaseQuery.updateMostRecent(currentEpoch);
165 }
166
167 for (final Result result : results) {
168 final String id = Bytes.toString(result.getValue(meta, Bytes.toBytes("i")));
169 final String directory = Bytes.toString(result.getValue(meta, Bytes.toBytes("d")));
170 final String stream = Bytes.toString(result.getValue(meta, Bytes.toBytes("s")));
171 final String host = Bytes.toString(result.getValue(meta, Bytes.toBytes("h")));
172 final String bucket = Bytes.toString(result.getValue(meta, Bytes.toBytes("b")));
173 final String path = Bytes.toString(result.getValue(meta, Bytes.toBytes("p")));
174 final long logtime = Bytes.toLong(result.getValue(meta, Bytes.toBytes("t")));
175 final long filesize = ULong.valueOf(Bytes.toLong(result.getValue(meta, Bytes.toBytes("fs")))).longValue();
176 final byte[] uncompressedFileSizeBytes = result.getValue(meta, Bytes.toBytes("ufs"));
177 final long uncompressedFileSize;
178
179 // uncompressed can be null, null values are replicated as empty byte arrays to hbase
180 if (uncompressedFileSizeBytes.length == 0) {
181 uncompressedFileSize = -1L;
182 }
183 else {
184 uncompressedFileSize = ULong.valueOf(Bytes.toLong(uncompressedFileSizeBytes)).longValue();
185 }
186
187 this
188 .add(
189 new BatchSlice(
190 new ArchiveS3ObjectMetadata(
191 id,
192 bucket,
193 path,
194 directory,
195 stream,
196 host,
197 logtime,
198 filesize,
199 uncompressedFileSize
200 )
201 )
202 );
203 }
204
205 return this;
206 }
