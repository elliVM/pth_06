87 public HBaseBatchSliceCollection processRange(final Offset start, final Offset end) {
88 this.clear(); // clear internal list
89 final long startOffsetLong = ((DatasourceOffset) start).getArchiveOffset().offset();
90 final long endOffsetLong = ((DatasourceOffset) end).getArchiveOffset().offset();
91 LOGGER.debug("processRange() start <{}> end <{}>", startOffsetLong, endOffsetLong);
92 final List<ScanRangeView> scanRangeViews = hBaseQuery.openViews();
93 final SynchronizedHourlyResults synchronizedHourlyResults = new SynchronizedHourlyResults(
94 scanRangeViews,
95 startOffsetLong
96 );
97 final List<Result> results = new ArrayList<>();
98 final long maxWeight = (long) quantumLength * numPartitions;
99 while (synchronizedHourlyResults.hasNext()) {
100 final List<Result> hourlyResults = synchronizedHourlyResults.nextHour();
101 final long currentEpoch = synchronizedHourlyResults.currentEpoch();
102 if (!hourlyResults.isEmpty()) {
103 final BatchSizeLimit batchSizeLimit = new BatchSizeLimit(maxWeight, totalObjectCountLimit);
104 for (final Result hourlyResult : hourlyResults) {
105 final long fileSize = Bytes.toLong(hourlyResult.getValue(meta, Bytes.toBytes("fs")));
106 final float hourlyResultEstimatedFileSize = fileSize * fileCompressionRatio / 1024 / 1024
107 / processingSpeed;
108 batchSizeLimit.add(hourlyResultEstimatedFileSize);
109 if (!batchSizeLimit.isOverLimit()) {
110 results.add(hourlyResult);
111 }
112 else {
113 LOGGER
114 .info(
115 "Hourly results were over batch size limit with <{}> logfiles, ignoring rest of the results",
116 hourlyResult.size()
117 );
118 break;
119 }
120 }
121 }
122 hBaseQuery.updateLatest(currentEpoch);
123 }
124
125 for (final Result result : results) {
126 final String id = Bytes.toString(result.getValue(meta, Bytes.toBytes("i")));
127 final String directory = Bytes.toString(result.getValue(meta, Bytes.toBytes("d")));
128 final String stream = Bytes.toString(result.getValue(meta, Bytes.toBytes("s")));
129 final String host = Bytes.toString(result.getValue(meta, Bytes.toBytes("h")));
130 final String bucket = Bytes.toString(result.getValue(meta, Bytes.toBytes("b")));
131 final String path = Bytes.toString(result.getValue(meta, Bytes.toBytes("p")));
132 final long logtime = Bytes.toLong(result.getValue(meta, Bytes.toBytes("t")));
133 final long filesize = ULong.valueOf(Bytes.toLong(result.getValue(meta, Bytes.toBytes("fs")))).longValue();
134 final byte[] uncompressedFileSizeBytes = result.getValue(meta, Bytes.toBytes("ufs"));
135 final long uncompressedFileSize;
136
137 // uncompressed can be null, null values are replicated as empty byte arrays to hbase
138 if (uncompressedFileSizeBytes.length == 0) {
139 uncompressedFileSize = -1L;
140 }
141 else {
142 uncompressedFileSize = ULong.valueOf(Bytes.toLong(uncompressedFileSizeBytes)).longValue();
143 }
144
145 this
146 .add(
147 new BatchSlice(
148 new ArchiveS3ObjectMetadata(
149 id,
150 bucket,
151 path,
152 directory,
153 stream,
154 host,
155 logtime,
156 filesize,
157 uncompressedFileSize
158 )
159 )
160 );
161 }
162
163 return this;
164 }
