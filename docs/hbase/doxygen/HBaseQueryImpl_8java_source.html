156 @Override
157 public long earliest() {
158 final List<ScanPlan> rangeList = scanPlanCollection.asList();
159 if (rangeList.isEmpty()) {
160 throw new IllegalArgumentException(
161 "Cannot determine earliest value, no scan plans were available for query - index might not exist"
162 );
163 }
164 long earliest = Long.MAX_VALUE;
165 for (final ScanPlan range : rangeList) {
166 earliest = Math.min(earliest, range.earliest());
167 }
168 LOGGER.debug("earliest() called with offset <{}>", earliest);
169 return earliest;
170 }
171
177 @Override
178 public long latest() {
179 long startOffset;
180 if (mostRecentCommitedOffset == Long.MIN_VALUE) {
181 startOffset = earliest();
182 }
183 else {
184 startOffset = mostRecentCommitedOffset;
185 }
186 if (!isOpen()) {
187 LOGGER.info("latest() called advancing query results from start offset <{}>", startOffset);
188 open(startOffset);
189 }
190 final long quantumLength = config.batchConfig.quantumLength;
191 final long numPartitions = config.batchConfig.numPartitions;
192 final long maxWeight = quantumLength * numPartitions;
193 final long totalObjectCountLimit = config.batchConfig.totalObjectCountLimit;
194 final Timer.Context timerCtx = metricRegistry.timer("ArchiveDatabaseLatency").time();
195 final BatchSizeLimit batchSizeLimit = new BatchSizeLimit(maxWeight, totalObjectCountLimit);
196 final long previousBatchOffset;
197 if (!limitedResults.isStub()) {
198 previousBatchOffset = limitedResults.latest();
199 }
200 else {
201 previousBatchOffset = mostRecentOffset();
202 }
203 // update the current batch results
204 this.limitedResults = new BatchSizeLimitedResults(
205 hourlySlices,
206 batchSizeLimit,
207 config,
208 previousBatchOffset,
209 new MetricRegistry()
210 );
211 // update metrics
212 long rows = limitedResults.results().size();
213 final long latencyNs = timerCtx.stop();
214 if (rows != 0) {
215 metricRegistry.histogram("ArchiveDatabaseLatencyPerRow").update(latencyNs / rows);
216 }
217 metricRegistry.counter("ArchiveDatabaseRowCount").inc(rows);
218 final long latest = limitedResults.latest();
219 return latest;
220 }
221

226 }
