89 @Override
90 public long earliest() {
91 final List<ScanPlan> rangeList = scanPlanCollection.asList();
92 final long earliest;
93 if (rangeList.isEmpty()) {
94 earliest = ZonedDateTime.now().minusHours(24).toEpochSecond();
95 }
96 else {
97 long min = Long.MAX_VALUE;
98 for (final ScanPlan range : rangeList) {
99 min = Math.min(min, range.earliest());
100 }
101 earliest = min;
102 }
103 return earliest;
104 }
105
106 @Override
107 public long latest() {
108 final long latest;
109 if (latestCommited == Long.MIN_VALUE) {
110 latest = earliest();
111 }
112 else {
113 latest = latestCommited + 3600L;
114 }
115 return latest;
116 }
117
118 @Override
119 public long mostRecentOffset() {

121 }

123 @Override
124 public CustomTaskMetric[] currentDatabaseMetrics() {
125 // TODO update values
126 final Snapshot latencySnapshot = metricRegistry.histogram("ArchiveDatabaseLatencyPerRow").getSnapshot();
127 return new CustomTaskMetric[] {
128 new TaskMetric("ArchiveDatabaseRowCount", metricRegistry.counter("ArchiveDatabaseRowCount").getCount()),
129 new TaskMetric("ArchiveDatabaseRowMaxLatency", latencySnapshot.getMax()),
130 new TaskMetric("ArchiveDatabaseRowAvgLatency", (long) latencySnapshot.getMean()),
131 new TaskMetric("ArchiveDatabaseRowMinLatency", latencySnapshot.getMin()),
132 };
133 }
