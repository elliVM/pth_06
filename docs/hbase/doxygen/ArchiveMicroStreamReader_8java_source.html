142 @Override
143 public Offset initialOffset() {
144 // archive only: subtract 3600s (1 hour) from earliest to return first row (start exclusive)
145 final DatasourceOffset rv;
146 final boolean archiverEnabled = !hBaseQuery.isStub() || !archiveQuery.isStub();
147 final boolean kafkaEnabled = !kafkaQuery.isStub();
148 final boolean bothEnabled = archiverEnabled && kafkaEnabled;
149
150 if (bothEnabled) {
151 final LongOffset archiveInitialOffset;
152 if (!hBaseQuery.isStub()) {
153 archiveInitialOffset = new LongOffset(hBaseQuery.earliest());
154 }
155 else { // use hbase if enabled
156 archiveInitialOffset = new LongOffset(archiveQuery.getInitialOffset() - 3600L);
157 }
158 rv = new DatasourceOffset(archiveInitialOffset, new KafkaOffset(kafkaQuery.getBeginningOffsets(null)));
159 }
160 else if (archiverEnabled) {
161 if (hBaseQuery.isStub()) {
162 rv = new DatasourceOffset(new LongOffset(archiveQuery.getInitialOffset() - 3600L));
163 }
164 else { // use hbase if enabled
165 rv = new DatasourceOffset(new LongOffset(hBaseQuery.earliest()));
166 }
167 }
168 else if (kafkaEnabled) {
169 rv = new DatasourceOffset(new KafkaOffset(kafkaQuery.getBeginningOffsets(null)));
170 }
171 else {
172 throw new IllegalStateException("no datasources enabled, can't get initial offset");
173 }
174
175 LOGGER.debug("offset[initial]= {}", rv);
176 return rv;
177 }

256 @Override
257 public InputPartition[] planInputPartitions(final Offset start, final Offset end) {
258 final List<InputPartition> inputPartitions = new LinkedList<>();
259
260 final Batch currentBatch = new Batch(config, archiveQuery, kafkaQuery, hBaseQuery).processRange(start, end);
261
262 for (final LinkedList<BatchSlice> taskObjectList : currentBatch) {
263
264 // archive tasks
265 final LinkedList<ArchiveS3ObjectMetadata> archiveTaskList = new LinkedList<>();
266 for (final BatchSlice batchSlice : taskObjectList) {
267 if (batchSlice.type.equals(BatchSlice.Type.ARCHIVE)) {
268 archiveTaskList.add(batchSlice.archiveS3ObjectMetadata);
269 }
270 }
271
272 if (!archiveTaskList.isEmpty()) {
273 inputPartitions
274 .add(
275 new ArchiveMicroBatchInputPartition(
276 config.archiveConfig.s3EndPoint,
277 config.archiveConfig.s3Identity,
278 config.archiveConfig.s3Credential,
279 archiveTaskList,
280 config.auditConfig.query,
281 config.auditConfig.reason,
282 config.auditConfig.user,
283 config.auditConfig.pluginClassName,
284 config.archiveConfig.skipNonRFC5424Files
285 )
286 );
287 }
288
289 // kafka tasks
290 for (final BatchSlice batchSlice : taskObjectList) {
291 if (batchSlice.type.equals(BatchSlice.Type.KAFKA)) {
292 inputPartitions
293 .add(
294 new KafkaMicroBatchInputPartition(
295 config.kafkaConfig.executorOpts,
296 batchSlice.kafkaTopicPartitionOffsetMetadata.topicPartition,
297 batchSlice.kafkaTopicPartitionOffsetMetadata.startOffset,
298 batchSlice.kafkaTopicPartitionOffsetMetadata.endOffset,
299 config.kafkaConfig.executorConfig,
300 config.kafkaConfig.skipNonRFC5424Records
301 )
302 );
303 }
304 }
305 }
306
307 return inputPartitions.toArray(new InputPartition[0]);
308 }
