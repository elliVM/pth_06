58public final class FilterGroup {
59
60 private final List<XMLValueExpression> hostList;
61 private final List<XMLValueExpression> sourceTypeList;
62
63 public FilterGroup(final ClassifiedXMLValueExpressions classifiedXMLValueExpressions) {
64 this(classifiedXMLValueExpressions.hostList(), classifiedXMLValueExpressions.sourceTypeList());
65 }
66
67 public FilterGroup(final List<XMLValueExpression> hostList, final List<XMLValueExpression> sourceTypeList) {
68 this.hostList = hostList;
69 this.sourceTypeList = sourceTypeList;
70 }
71
72 public FilterList filterList() {
73 final FilterList filterList = new FilterList();
74
75 for (XMLValueExpression hostExpression : hostList) {
76 final String value = hostExpression.value();
77 final String operation = hostExpression.operation();
78 final CompareOperator operator;
79 if ("EQUALS".equalsIgnoreCase(operation)) {
80 operator = CompareOperator.EQUAL;
81 }
82 else if ("NOT_EQUALS".equalsIgnoreCase(operation)) {
83 operator = CompareOperator.NOT_EQUAL;
84 }
85 else {
86 throw new IllegalArgumentException("Unsupported operation <" + operation + ">");
87 }
88 final Filter hostFilter = new SingleColumnValueFilter(
89 Bytes.toBytes("meta"), // column family
90 Bytes.toBytes("h"), // host
91 operator,
92 new RegexStringComparator(value)
93 );
94 filterList.addFilter(hostFilter);
95 }
96
97 for (final XMLValueExpression sourceTypeExpression : sourceTypeList) {
98 final String value = sourceTypeExpression.value();
99 final String operation = sourceTypeExpression.operation();
100 final CompareOperator operator;
101 if ("EQUALS".equalsIgnoreCase(operation)) {
102 operator = CompareOperator.EQUAL;
103 }
104 else if ("NOT_EQUALS".equalsIgnoreCase(operation)) {
105 operator = CompareOperator.NOT_EQUAL;
106 }
107 else {
108 throw new IllegalArgumentException("Unsupported operation <" + operation + ">");
109 }
110 final Filter sourceTypeFilter = new SingleColumnValueFilter(
111 Bytes.toBytes("meta"), // column family
112 Bytes.toBytes("s"), // stream
113 operator,
114 new RegexStringComparator(value)
115 );
116 filterList.addFilter(sourceTypeFilter);
117 }
118
119 return filterList;
120 }
