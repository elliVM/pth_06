138 private void collectScanPlans(final DSLContext ctx, final Expression expression) {
139 if (expression.isLogical()) {
140 final List<Expression> children = expression.asLogical().children();
141 for (final Expression child : children) {
142 if (child.isLogical()) {
143 collectScanPlans(ctx, child);
144 }
145 }
146 final ScanGroupExpression scanGroupExpression = new ScanGroupExpression(ctx, expression.asLogical());
147 scanPlans.addAll(scanGroupExpression.value());
148 }
