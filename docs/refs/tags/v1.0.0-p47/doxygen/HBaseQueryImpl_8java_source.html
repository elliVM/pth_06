145 @Override
146 public long earliest() {
147 final List<ScanPlan> rangeList = scanPlanCollection.asList();
148 if (rangeList.isEmpty()) {
149 throw new IllegalArgumentException(
150 "Cannot determine earliest value, no scan plans were available for query - index might not exist"
151 );
152 }
153 long earliest = Long.MAX_VALUE;
154 for (final ScanPlan range : rangeList) {
155 earliest = Math.min(earliest, range.earliest());
156 }
157 LOGGER.debug("earliest() called with offset <{}>", earliest);
158 return earliest;
159 }
160
166 @Override
167 public long latest() {
168 long startOffset;
169 if (mostRecentCommitedOffset == Long.MIN_VALUE) {
170 startOffset = earliest();
171 }
172 else {
173 startOffset = mostRecentCommitedOffset;
174 }
175 if (!isOpen()) {
176 LOGGER.info("latest() called advancing query results from start offset <{}>", startOffset);
177 open(startOffset);
178 }
179 final byte[] columnFamilyBytes = Bytes.toBytes("meta");
180 final long quantumLength = config.batchConfig.quantumLength;
181 final long numPartitions = config.batchConfig.numPartitions;
182 final long maxWeight = quantumLength * numPartitions;
183 final long totalObjectCountLimit = config.batchConfig.totalObjectCountLimit;
184 final float fileCompressionRatio = config.batchConfig.fileCompressionRatio;
185 final float processingSpeed = config.batchConfig.processingSpeed;
186 final BatchSizeLimit batchSizeLimit = new BatchSizeLimit(maxWeight, totalObjectCountLimit);
187
188 this.batchResult = new ArrayList<>(); // replace last batch results
189 while (!batchSizeLimit.isOverLimit() && hasNext()) {
190 final List<Result> nextHourResults = hourlySlices.nextHour();
191 for (final Result hourlyResult : nextHourResults) {
192 final long fileSize = Bytes.toLong(hourlyResult.getValue(columnFamilyBytes, Bytes.toBytes("fs")));
193 final float hourlyResultEstimatedFileSize = fileSize * fileCompressionRatio / 1024 / 1024
194 / processingSpeed;
195 batchSizeLimit.add(hourlyResultEstimatedFileSize);
196 mostRecentOffset = new EpochFromRowKey(hourlyResult.getRow()).epoch();
197 batchResult.add(hourlyResult);
198 if (batchSizeLimit.isOverLimit()) {
199 LOGGER.info("Batch limit reached at offset at <{}>", mostRecentOffset);
200 break;
201 }
202 }
203 }
204 return mostRecentOffset;
205 }
