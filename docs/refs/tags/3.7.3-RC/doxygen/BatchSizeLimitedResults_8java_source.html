124 @Override
125 public List<Result> results() {
126 if (results.isEmpty()) {
127 long latencyNs;
128 final byte[] columnFamilyBytes = Bytes.toBytes("meta");
129 try (final Timer.Context timerCtx = metricRegistry.timer("ArchiveDatabaseLatency").time()) {
130 while (!limit.isOverLimit() && !slices.isStub() && slices.hasNext()) {
131 final List<Result> nextHourResults = slices.nextHour();
132 for (final Result hourlyResult : nextHourResults) {
133 final long fileSize = Bytes
134 .toLong(hourlyResult.getValue(columnFamilyBytes, Bytes.toBytes("fs")));
135 final float hourlyResultEstimatedFileSize = fileSize * fileCompressionRatio / 1024 / 1024
136 / processingSpeed;
137 limit.add(hourlyResultEstimatedFileSize);
138 results.add(hourlyResult);
139 if (limit.isOverLimit()) {
140 break;
141 }
142 }
143 }
144 latencyNs = timerCtx.stop();
145 }
146 // update metrics
147 final long rows = results.size();
148 if (rows > 0) {
149 metricRegistry.histogram("ArchiveDatabaseLatencyPerRow").update(latencyNs / rows);
150 }
151 metricRegistry.counter("ArchiveDatabaseRowCount").inc(rows);
152 }
153 return results;
154 }
155
156 @Override
157 public long latest() {
158 results();
159 long latest;
160 if (!results.isEmpty()) {
161 Result lastResult = results.get(results().size() - 1);
162 latest = new EpochFromRowKey(lastResult.getRow()).epoch();
163 }
164 else {
165 latest = startingOffset;
166 }
167 return latest;
168 }

172 return false;
