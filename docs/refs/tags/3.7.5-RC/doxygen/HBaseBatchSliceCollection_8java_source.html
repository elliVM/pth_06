111 public HBaseBatchSliceCollection processRange(final Offset start, final Offset end) {
112 this.clear(); // clear internal list
113 final long startOffsetLong = ((DatasourceOffset) start).getArchiveOffset().offset();
114 final long endOffsetLong = ((DatasourceOffset) end).getArchiveOffset().offset();
115 LOGGER.info("HBaseQuery process range <{}>-<{}>", startOffsetLong, endOffsetLong);
116 if (!hBaseQuery.isOpen()) {
117 hBaseQuery.open(startOffsetLong);
118 }
119
120 final List<Result> results = hBaseQuery.currentBatch();
121
122 for (final Result result : results) {
123 final String id = Bytes.toString(result.getValue(meta, Bytes.toBytes("i")));
124 final String directory = Bytes.toString(result.getValue(meta, Bytes.toBytes("d")));
125 final String stream = Bytes.toString(result.getValue(meta, Bytes.toBytes("s")));
126 final String host = Bytes.toString(result.getValue(meta, Bytes.toBytes("h")));
127 final String bucket = Bytes.toString(result.getValue(meta, Bytes.toBytes("b")));
128 final String path = Bytes.toString(result.getValue(meta, Bytes.toBytes("p")));
129 final long logtime = Bytes.toLong(result.getValue(meta, Bytes.toBytes("t")));
130 final long filesize = ULong.valueOf(Bytes.toLong(result.getValue(meta, Bytes.toBytes("fs")))).longValue();
131 final byte[] uncompressedFileSizeBytes = result.getValue(meta, Bytes.toBytes("ufs"));
132 final long uncompressedFileSize;
133
134 // uncompressed can be null, null values are replicated as empty byte arrays to hbase
135 if (uncompressedFileSizeBytes.length == 0) {
136 uncompressedFileSize = -1L;
137 }
138 else {
139 uncompressedFileSize = ULong.valueOf(Bytes.toLong(uncompressedFileSizeBytes)).longValue();
140 }
141
142 this
143 .add(
144 new BatchSlice(
145 new ArchiveS3ObjectMetadata(
146 id,
147 bucket,
148 path,
149 directory,
150 stream,
151 host,
152 logtime,
153 filesize,
154 uncompressedFileSize
155 )
156 )
157 );
158 }
159
160 return this;
161 }
