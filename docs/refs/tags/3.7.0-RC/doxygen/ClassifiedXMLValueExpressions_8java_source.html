82 private List<XMLValueExpression> expressionListFilteredByTag(final Expression.Tag tag) {
83 final List<XMLValueExpression> list = new ArrayList<>();
84 for (final Expression expression : expressions) {
85 if (expression.tag().equals(tag) && expression.isLeaf()) {
86 list.add((XMLValueExpression) expression);
87 }
88 }
89 return list;
