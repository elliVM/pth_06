77 public Expression asAST() {
78 final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
79 final DocumentBuilder builder;
80 try {
81 builder = factory.newDocumentBuilder();
82 }
83 catch (final ParserConfigurationException e) {
84 throw new RuntimeException("Error creating document builder: " + e.getMessage());
85 }
86 final Document document;
87 try {
88 document = builder.parse(new InputSource(new StringReader(xmlString)));
89 }
90 catch (final IOException | SAXException e) {
91 throw new RuntimeException("Error parsing XML to elements: " + e.getMessage());
92 }
93 final Element root = document.getDocumentElement();
94 return elementToNode(root);
95 }

97 private Expression elementToNode(final Element element) {
98 final String tagName = element.getTagName();
99 LOGGER.info("Incoming element <{}>", tagName);
100 final Expression result;
101 switch (tagName.toLowerCase()) {
102 case "and":
103 final List<Expression> andExpressionChildren = visitLogical(element);
104 result = new AndExpression(andExpressionChildren);
105 break;
106 case "or":
107 final List<Expression> orExpressionChildren = visitLogical(element);
108 result = new OrExpression(orExpressionChildren);
109 break;
110 case "index":
111 result = visitLeaf(element, Expression.Tag.INDEX);
112 break;
113 case "host":
114 result = visitLeaf(element, Expression.Tag.HOST);
115 break;
116 case "sourcetype":
117 result = visitLeaf(element, Expression.Tag.SOURCETYPE);
118 break;
119 case "earliest":
120 case "index_earliest":
121 result = visitLeaf(element, Expression.Tag.EARLIEST);
122 break;
123 case "latest":
124 case "index_latest":
125 result = visitLeaf(element, Expression.Tag.LATEST);
126 break;
127 case "indexstatement":
128 result = visitLeaf(element, Expression.Tag.INDEXSTATEMENT);
129 break;
130 default:
131 throw new IllegalArgumentException("Unsupported element <" + tagName + ">");
132 }
133 return result;
134 }
135
136 private Expression visitLeaf(final Element element, final Expression.Tag tag) {
137 final String value = element.getAttribute("value");
138 final String operation = element.getAttribute("operation");
139 return new XMLValueExpressionImpl(value, operation, tag);
140 }
141
142 private List<Expression> visitLogical(final Element element) {
143 final List<Expression> expressions = new ArrayList<>();
144 final NodeList childNodes = element.getChildNodes();
145 for (int i = 0; i < childNodes.getLength(); i++) {
146 final Node node = childNodes.item(i);
147 if (node instanceof Element) {
148 final Element nodeElement = (Element) node;
149 // recursive call
150 final Expression exp = elementToNode(nodeElement);
151 expressions.add(exp);
152 }
153 else {
154 throw new IllegalArgumentException("Element children contained a non Element node");
155 }
156 }
157 if (expressions.isEmpty()) {
158 throw new IllegalStateException("Logical expression had no children");
159 }
160 return expressions;
161 }
