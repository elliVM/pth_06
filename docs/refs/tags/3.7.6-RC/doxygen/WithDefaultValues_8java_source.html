92 private Expression recursiveWithDefaultsExpression(final Expression expression) {
93 final Expression result;
94 if (!expression.isLogical()) {
95 result = expression;
96 }
97 else if (expression.tag().equals(Expression.Tag.AND)) {
98 result = andWithDefaults(expression);
99 }
100 else {
101 result = orWithDefaults(expression);
102 }
103 return result;
104 }

106 private Expression andWithDefaults(final Expression expression) {
107 final List<Expression> andChildren = new ArrayList<>();
108 for (final Expression child : expression.asLogical().children()) {
109 andChildren.add(recursiveWithDefaultsExpression(child));
110 }
111 final Set<Expression.Tag> tags = expressionTags(new AndExpression(andChildren));
112 if (!tags.contains(Expression.Tag.INDEX)) {
113 andChildren.add(new XMLValueExpressionImpl("*", "EQUALS", Expression.Tag.INDEX));
114 }
115 if (!tags.contains(Expression.Tag.EARLIEST)) {
116 andChildren
117 .add(new XMLValueExpressionImpl(String.valueOf(ZonedDateTime.now().minusHours(defaultMinusHours).toEpochSecond()), "EQUALS", Expression.Tag.EARLIEST));
118 }
119 if (!tags.contains(Expression.Tag.LATEST)) {
120 andChildren
121 .add(new XMLValueExpressionImpl(String.valueOf(ZonedDateTime.now().toEpochSecond()), "EQUALS", Expression.Tag.LATEST));
122 }
123 return new AndExpression(andChildren);
124 }
125
126 private Expression orWithDefaults(final Expression expression) {
127 final List<Expression> orChildrenWithDefaults = new ArrayList<>();
128 for (final Expression child : expression.asLogical().children()) {
129 final Expression childWithDefaults = recursiveWithDefaultsExpression(child);
130
131 // Add defaults to the child if missing
132 final Set<Expression.Tag> tags = expressionTags(childWithDefaults); // your helper or similar
133 final List<Expression> childExpressions = new ArrayList<>();
134 if (childWithDefaults.isLogical() && childWithDefaults.tag().equals(Expression.Tag.AND)) {
135 childExpressions.addAll(childWithDefaults.asLogical().children());
136 }
137 else {
138 childExpressions.add(childWithDefaults);
139 }
140
141 if (!tags.contains(Expression.Tag.INDEX)) {
142 childExpressions.add(new XMLValueExpressionImpl("*", "EQUALS", Expression.Tag.INDEX));
143 }
144 if (!tags.contains(Expression.Tag.EARLIEST)) {
145 childExpressions
146 .add(new XMLValueExpressionImpl(String.valueOf(ZonedDateTime.now().minusHours(defaultMinusHours).toEpochSecond()), "EQUALS", Expression.Tag.EARLIEST));
147 }
148 if (!tags.contains(Expression.Tag.LATEST)) {
149 childExpressions
150 .add(new XMLValueExpressionImpl(String.valueOf(ZonedDateTime.now().toEpochSecond()), "EQUALS", Expression.Tag.LATEST));
151 }
152
153 // Wrap child expressions into an AND (or single expression if only one)
154 final Expression newChild = (childExpressions.size() == 1) ? childExpressions
155 .get(0) : new AndExpression(childExpressions);
156 orChildrenWithDefaults.add(newChild);
157 }
158 return new OrExpression(orChildrenWithDefaults);
159 }

161 private Set<Expression.Tag> expressionTags(final Expression expr) {
162 final Set<Expression.Tag> tags = new HashSet<>();
163 if (expr.isLogical()) {
164 for (Expression child : expr.asLogical().children()) {
165 tags.addAll(expressionTags(child));
166 }
167 }
168 else {
169 tags.add(expr.tag());
170 }
171 return tags;
172 }
